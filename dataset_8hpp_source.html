128  make_batchset(DocPtr doc, InputReaderPtr reader, size_t batch_size=32,

150 
151  // shorter first
152  std::sort(keys.begin(), keys.end(), Sample::compare);
153 
154  // merge samples into minibatches
155  std::vector<std::vector<Sample>> batchset;
156  size_t start_id = 0;
157  while (true) {
158  const auto& start = keys[start_id];
159  auto factor = std::max<size_t>(start.ilen / max_length_in, start.olen / max_length_out);
160  auto b = std::max<size_t>(1, batch_size / (1 + factor));
161  auto end_id = std::min<size_t>(keys.size(), start_id + b);
162  std::vector<Sample> mb(keys.begin() + start_id, keys.begin() + end_id);
163  batchset.push_back(mb);
164  if (end_id == keys.size() || batchset.size() > max_num_batches) break;
165  start_id = end_id;
166  }
167  return batchset;

171  struct MiniBatch {
172  MiniBatch(const std::vector<Sample>& minibatch) {
173  this->input_lengths.reserve(minibatch.size());
174  this->target_lengths.reserve(minibatch.size());
175  std::int64_t max_ilen = 0, max_olen = 0;
176  for (const auto& sample : minibatch) {
177  this->input_lengths.push_back(sample.ilen);
178  this->target_lengths.push_back(sample.olen);
179  if (sample.ilen > max_ilen) max_ilen = sample.ilen;
180  if (sample.olen > max_olen) max_olen = sample.olen;
181  }
182  auto mb_size = static_cast<std::int64_t>(minibatch.size());
183  // maybe the bug of at::Tensor, memory leaks unless this unique_ptr
184  // TODO make Variable().data unique_ptr
185  this->inputs = memory::make_unique<at::Tensor>(at::zeros({mb_size, max_ilen, minibatch.front().idim}));
186  this->targets = memory::make_unique<at::Tensor>(at::zeros({mb_size, max_olen}, at::kLong));
187  for (size_t batch_idx = 0; batch_idx < minibatch.size(); ++batch_idx) {
188  auto x = minibatch[batch_idx].input();
189  auto t = minibatch[batch_idx].target();
190  (*inputs)[batch_idx].slice(0, 0, x.size(0)) = x;
191  (*targets)[batch_idx].slice(0, 0, t.size(0)) = t;
192  }
193  }
194 
195  std::unique_ptr<at::Tensor> inputs, targets;
196  std::vector<std::int64_t> input_lengths, target_lengths;
197  };
